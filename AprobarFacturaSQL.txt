CREATE TABLE AJUSTES (

CREATE TABLE BONIFICACIONES (
   BONCODIGO            CHAR(5)              NOT NULL,

CREATE TABLE BXN (
   BONCODIGO            CHAR(5)              NOT NULL,

CREATE TABLE CLIENTES (
   CLICODIGO            CHAR(7)              NOT NULL,
   CLINOMBRE            CHAR(60)             NOT NULL,

CREATE TABLE COMPRAS (
